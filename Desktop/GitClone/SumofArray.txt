ALGORITHM sum_distinct_elements
VAR
    A : ARRAY [0..n1-1] OF INTEGER     // Set 1 (as array, may contain repeats)
    B : ARRAY [0..n2-1] OF INTEGER     // Set 2 (as array, may contain repeats)
    i, j : INTEGER
    found : BOOLEAN
    sum : INTEGER

BEGIN
    sum := 0

    // Add elements of A that are NOT in B
    FOR i := 0 TO n1-1 DO
        found := FALSE
        FOR j := 0 TO n2-1 DO
            IF A[i] = B[j] THEN
                found := TRUE
                BREAK
            END_IF
        END_FOR
        IF found = FALSE THEN
            sum := sum + A[i]
        END_IF
    END_FOR

    // Add elements of B that are NOT in A
    FOR i := 0 TO n2-1 DO
        found := FALSE
        FOR j := 0 TO n1-1 DO
            IF B[i] = A[j] THEN
                found := TRUE
                BREAK
            END_IF
        END_FOR
        IF found = FALSE THEN
            sum := sum + B[i]
        END_IF
    END_FOR

    Write(sum)
END
